350 template <typename numType>
351 Matrix<numType>* Matrix<numType>::gauss_jordan_inverse(double *det) const
352 {
353  if (nrows != ncols) {
354  throw runtime_error("Gauss-Jordan inversion: Can't invert Non-square matrix");
355  }
356  *det = 1.0f;
357  Matrix<numType> mat(nrows, ncols*2);
358  Matrix<numType> new_mat(nrows, ncols*2);
359 
360  int n = nrows;
361 
362  // Create matrix
363  for (int i=0 ; i<n ; i++) {
364  for (int j=0; j<n; j++) {
365  mat._data[i*2*n+j] = data[i*n+j];
366  }
367  mat._data[i*2*n+n+i] = 1;
368  }
369 
370  for (int k=0; k<n; k++) {
371  int i(k);
372  while (abs(mat._data[i*2*n+k]) < EPS) {
373  i++;
374  if (i==n) {
375  throw runtime_error("Non-invertible matrix");
376  }
377  }
378  *det *= mat._data[i*2*n+k];
379 
380  // if found > Exchange lines
381  if (i != k) {
382  mat.swap_lines(i, k);
383  }
384 
385  new_mat._data = mat._data;
386 
387  for (int j=0; j<2*n; j++) {
388  new_mat._data[k*2*n+j] /= mat._data[k*2*n+k];
389  }
390  for (i=0; i<n; i++) {
391  if (i != k) {
392  for (int j=0; j<2*n; j++) {
393  new_mat._data[i*2*n+j] -= mat._data[i*2*n+k] * new_mat._data[k*2*n+j];
394  }
395  }
396  }
397  mat._data = new_mat._data;
398  }
399 
400  Matrix<numType> *dst = new Matrix<numType>(nrows, ncols);
401  for (int i=0 ; i<n ; i++)
402  for (int j=0; j<n; j++)
403  dst->_data[i*n+j] = mat._data[i*2*n+n+j];
404  return dst;
405 }
