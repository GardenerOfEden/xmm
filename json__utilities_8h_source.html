162 template <typename T>
163 void json2array(JSONNode root, T* a, int n)
164 {
165  // Get Dimensions
166  if (root.type() != JSON_ARRAY)
167  throw JSONException("Wrong type: was expecting 'JSON_ARRAY'", root.name());
168  unsigned int i = 0;
169  for (JSONNode::const_iterator array_it = root.begin(); array_it != root.end(); ++array_it)
170  {
171  if (i >= n)
172  throw JSONException("JSON 2 Array: Index out of bounds");
173  if (array_it->type() != JSON_NUMBER)
174  throw JSONException("JSON 2 Vector: Wrong type");
175  a[i++] = array_it->as_int();
176  }
177 }
