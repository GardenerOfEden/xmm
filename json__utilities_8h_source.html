139 template <typename T>
140 void json2array(JSONNode root, T* a, int n)
141 {
142  // Get Dimensions
143  if (root.type() != JSON_ARRAY)
144  throw JSONException("Wrong type: was expecting 'JSON_ARRAY'", root.name());
145  unsigned int i = 0;
146  for (JSONNode::const_iterator array_it = root.begin(); array_it != root.end(); ++array_it)
147  {
148  if (i >= n)
149  throw JSONException("JSON 2 Array: Index out of bounds");
150  if (array_it->type() != JSON_NUMBER)
151  throw JSONException("JSON 2 Vector: Wrong type");
152  a[i++] = array_it->as_int();
153  }
154 }
