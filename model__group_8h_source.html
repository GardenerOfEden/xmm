448  pthread_create(&(training_threads[it->first]), NULL, &ModelType::train_func, &(it->second));

456  pthread_join(training_threads[it->first], NULL);

495  static void monitor_training(void *model, CALLBACK_FLAG state, void *extradata)
496  {
497  ModelGroup<ModelType> *thismodelgroup = (ModelGroup<ModelType> *)extradata;
498  if (state != TRAINING_RUN) {
499  Label label = ((ModelType *)model)->trainingSet->getPhraseLabel(0);
500  thismodelgroup->models_to_train_--;
501 #ifdef USE_PTHREAD
502  if (thismodelgroup->trainingCallbackFunction_)
503  thismodelgroup->training_threads.erase(label);
504 #endif
505  if (state == TRAINING_ERROR || state == TRAINING_ABORT) {
506  thismodelgroup->remove(label);
507  }
508  }
509  if (thismodelgroup->trainingCallbackFunction_)
510  thismodelgroup->trainingCallbackFunction_(model, state, thismodelgroup->trainingExtradata_);
511  }

552  virtual void update_likelihood_results()
553  {
554  double maxLogLikelihood = 0.0;
555  double normconst_instant(0.0);
556  double normconst_smoothed(0.0);
557  int i(0);
558  for (model_iterator it=this->models.begin(); it != this->models.end(); ++it, ++i) {
559  results_instant_likelihoods[i] = it->second.results_instant_likelihood;
560  results_log_likelihoods[i] = it->second.results_log_likelihood;
561  results_normalized_likelihoods[i] = exp(results_log_likelihoods[i]);
562 
563  normconst_instant += results_instant_likelihoods[i];
564  normconst_smoothed += results_normalized_likelihoods[i];
565 
566  if (i == 0 || results_log_likelihoods[i] > maxLogLikelihood) {
567  maxLogLikelihood = results_log_likelihoods[i];
568  results_likeliest = it->first;
569  }
570  }
571 
572  i = 0;
573  for (model_iterator it=this->models.begin(); it != this->models.end(); ++it, ++i) {
574  results_normalized_likelihoods[i] /= normconst_smoothed;
575  results_normalized_instant_likelihoods[i] = results_instant_likelihoods[i] / normconst_instant;
576  }
577  }
