498  static void monitor_training(void *model, CALLBACK_FLAG state, void *extradata)
499  {
500  ModelGroup<ModelType> *thismodelgroup = (ModelGroup<ModelType> *)extradata;
501  if (state != TRAINING_RUN) {
502  Label label = ((ModelType *)model)->trainingSet->getPhraseLabel(0);
503  thismodelgroup->models_to_train_--;
504 #ifdef USE_PTHREAD
505  if (thismodelgroup->trainingCallbackFunction_)
506  thismodelgroup->training_threads.erase(label);
507 #endif
508  if (state == TRAINING_ERROR || state == TRAINING_ABORT) {
509  thismodelgroup->remove(label);
510  }
511  }
512  if (thismodelgroup->trainingCallbackFunction_)
513  thismodelgroup->trainingCallbackFunction_(model, state, thismodelgroup->trainingExtradata_);
514  }

555  virtual void update_likelihood_results()
556  {
557  double maxLogLikelihood = 0.0;
558  double normconst_instant(0.0);
559  double normconst_smoothed(0.0);
560  int i(0);
561  for (model_iterator it=this->models.begin(); it != this->models.end(); ++it, ++i) {
562  results_instant_likelihoods[i] = it->second.results_instant_likelihood;
563  results_log_likelihoods[i] = it->second.results_log_likelihood;
564  results_normalized_likelihoods[i] = exp(results_log_likelihoods[i]);
565 
566  normconst_instant += results_instant_likelihoods[i];
567  normconst_smoothed += results_normalized_likelihoods[i];
568 
569  if (i == 0 || results_log_likelihoods[i] > maxLogLikelihood) {
570  maxLogLikelihood = results_log_likelihoods[i];
571  results_likeliest = it->first;
572  }
573  }
574 
575  i = 0;
576  for (model_iterator it=this->models.begin(); it != this->models.end(); ++it, ++i) {
577  results_normalized_likelihoods[i] /= normconst_smoothed;
578  results_normalized_instant_likelihoods[i] = results_instant_likelihoods[i] / normconst_instant;
579  }
