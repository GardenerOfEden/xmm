481  virtual void joinTraining()
482  {
483  if (is_training()) {
484  while (training_threads.size() > 0) {
485  if (pthread_join(training_threads.begin()->second, NULL))
486  throw std::runtime_error("Error joining the training threads!");

593  {

595 #ifdef USE_PTHREAD

605  thismodelgroup->trainingCallbackFunction_(model, state, thismodelgroup->trainingExtradata_);
606  thismodelgroup->locked_ = false;

659  virtual void update_likelihood_results()
660  {
661  check_training();
662  double maxLogLikelihood = 0.0;
663  double normconst_instant(0.0);
664  double normconst_smoothed(0.0);
665  int i(0);
666  for (model_iterator it=this->models.begin(); it != this->models.end(); ++it, ++i) {
667  results_instant_likelihoods[i] = it->second.results_instant_likelihood;
668  results_log_likelihoods[i] = it->second.results_log_likelihood;
669  results_normalized_likelihoods[i] = exp(results_log_likelihoods[i]);
670 
671  normconst_instant += results_instant_likelihoods[i];
672  normconst_smoothed += results_normalized_likelihoods[i];
673 
674  if (i == 0 || results_log_likelihoods[i] > maxLogLikelihood) {
675  maxLogLikelihood = results_log_likelihoods[i];
676  results_likeliest = it->first;
677  }
678  }
679 
680  i = 0;
681  for (model_iterator it=this->models.begin(); it != this->models.end(); ++it, ++i) {
682  results_normalized_likelihoods[i] /= normconst_smoothed;
683  results_normalized_instant_likelihoods[i] = results_instant_likelihoods[i] / normconst_instant;
684  }
685  }
